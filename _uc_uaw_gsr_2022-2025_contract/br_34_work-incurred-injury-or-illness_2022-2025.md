1<span class="tooltip"><span class="tooltip-text">{% assign tooltip_term = site.data.union_glossary | where: "Term", "loco parentis" %}{{ tooltip_term[0]["Meaning"] }}</span></span><span class="tooltip"><span class="tooltip-text">{% assign tooltip_term = site.data.union_glossary | where: "Term", "retaliation" %}{{ tooltip_term[0]["Meaning"] }}</span></span>. ## E. MODIFIED DUTY
